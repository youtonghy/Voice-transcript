// å¤„ç†Pythonè¿›ç¨‹è¾“å‡ºçš„å‡½æ•?function processPythonOutput(data) {

  let escapeNext = false;
  let currentMessage = '';
  
  for (let i = 0; i < pythonBuffer.length; i++) {

        if (braceCount === 0) {
          // æ‰¾åˆ°å®Œæ•´çš„JSONæ¶ˆæ¯

function restartPythonServiceAfterStop() {

    return;
  }

  restartingPython = true;
  try {
    console.log('å‘é€å…³é—­å‘½ä»¤ä»¥ä¼˜é›…é€€å‡ºè½¬å†™æœåŠ?);
    // å°è¯•ä¼˜é›…å…³é—­

  }

  let closed = false;

    closed = true;
    console.log('æ—§è½¬å†™æœåŠ¡å·²é€€å‡ºï¼Œå‡†å¤‡é‡å¯');
    pythonProcess = null;
    setTimeout(() => {
      const ok = startPythonService();

    }, 500);
  };

  // ä¸€æ¬¡æ€§ç›‘å¬å…³é—?  const closeHandler = (code, signal) => {
    console.log('æ”¶åˆ°è½¬å†™æœåŠ¡å…³é—­äº‹ä»¶(é‡å¯æµç¨‹):', code, signal);
    if (pythonProcess) {
      pythonProcess.removeListener('close', closeHandler);

  if (pythonProcess) {
    pythonProcess.once('close', closeHandler);
  }

  // è¶…æ—¶å¼ºåˆ¶å…³é—­
  setTimeout(() => {

        outputBuffer = lines.pop() || ''; // ä¿ç•™æœ€åä¸€è¡Œï¼ˆå¯èƒ½ä¸å®Œæ•´ï¼‰
        
        lines.forEach(line => {
          line = line.trim();
          if (!line) return;
          
          // å°è¯•è§£æJSONæ¶ˆæ¯
          try {
            const message = JSON.parse(line);
            if (mainWindow) {
              mainWindow.webContents.send('media-progress', message);
            }
            if (message.type === 'result') {
              hasResults = true;
            }
          } catch (e) {
            // ä¸æ˜¯JSONï¼Œä½œä¸ºæ™®é€šæ—¥å¿—å¤„ç?            if (line.includes('æ®µè½') || line.includes('è½¬å†™') || line.includes('ç¿»è¯‘') || line.includes('è¿›åº¦') || line.includes('å®Œæˆ')) {
              if (mainWindow) {
                mainWindow.webContents.send('media-progress', {
                  type: 'progress',
                  message: line,
                  progress: 50
                });
              }
              if (line.includes('è½¬å†™å®Œæˆ') || line.includes('ç¿»è¯‘å®Œæˆ')) {
                hasResults = true;
              }
            }
          }
        });
      });

      mediaTranscribeProcess.stderr.on('data', (data) => {
        const chunk = data.toString('utf8');
        stderrBuffer += chunk;
        
        // æŒ‰è¡Œå¤„ç†ï¼Œè¿‡æ»?FFmpeg çš„æ­£å¸¸æ—¥å¿—ï¼Œä¿ç•™æ˜æ˜¾é”™è¯¯
        const lines = chunk.split('\n');
        lines.forEach(rawLine => {
          const line = String(rawLine || '').trim();
          if (!line) return;

          const isError = /(error|failed|traceback|not found|invalid)/i.test(line);
          if (isError) {
            console.error('åª’ä½“è½¬å†™é”™è¯¯:', line);
            if (mainWindow) {
              mainWindow.webContents.send('media-progress', {
                type: 'error',
                message: line
              });
            }
          } else {
            // å°†éè‡´å‘½ä¿¡æ¯ä½œä¸ºè¿›åº¦/æ—¥å¿—å±•ç¤ºï¼Œé¿å…é€ æˆâ€œé”™è¯¯â€è¯¯æŠ?            if (mainWindow) {
              mainWindow.webContents.send('media-progress', {
                type: 'progress',
                message: line,
                progress: 50
              });
            }
          }
        });
      });

      mediaTranscribeProcess.on('error', (error) => {
        console.error('åª’ä½“è½¬å†™è¿›ç¨‹å¯åŠ¨å¤±è´¥:', error);
        reject(new Error(`è¿›ç¨‹å¯åŠ¨å¤±è´¥: ${error.message}`));
      });

      mediaTranscribeProcess.on('close', (code, signal) => {
        console.log(`åª’ä½“è½¬å†™è¿›ç¨‹é€€å‡ºï¼Œä»£ç : ${code}, ä¿¡å·: ${signal}`);
        
        if (code === 0) {

          if (mainWindow) {

              type: 'complete',
              message: 'å¤„ç†å®Œæˆ',
              progress: 100
            });
          }
          resolve({
            success: true,
            hasResults: hasResults
          });
        } else {
          // å¤„ç†å¤±è´¥
          const tail = stderrBuffer.split(/\r?\n/).filter(Boolean).slice(-10).join('\n');
          const errorMsg = `å¤„ç†å¤±è´¥ï¼Œé€€å‡ºä»£ç ? ${code}${tail ? `\nè¯¦æƒ…: ${tail}` : ''}`;
          if (mainWindow) {
            mainWindow.webContents.send('media-progress', {
              type: 'error',
              message: errorMsg
            });
          }
          reject(new Error(errorMsg));
        }
        
        mediaTranscribeProcess = null;
      });

      // è®¾ç½®è¶…æ—¶ï¼?0åˆ†é’Ÿï¼?      setTimeout(() => {
        if (mediaTranscribeProcess) {
          mediaTranscribeProcess.kill();
          reject(new Error('å¤„ç†è¶…æ—¶'));
        }
      }, 30 * 60 * 1000);

    } catch (error) {
      console.error('å¯åŠ¨åª’ä½“è½¬å†™è¿›ç¨‹å¤±è´¥:', error);
      reject(error);
    }
  });
}
